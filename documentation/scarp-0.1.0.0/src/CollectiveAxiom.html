-- | only a test function
checkAllEqual :: (Eq a, Ord a, Show a, Num a) => [HChoice3] -> HChoice3 -> Bool
checkAllEqual parents (y0,_,_) = and [x0 == y0 | (x0,_,_) <- parents]

checkCollectiveRationality :: (Eq a, Ord a, Show a, Num a) => [HChoice3] -> Bool
checkCollectiveRationality ps = all (\f -> f r0H rH r0M rM r0F rF r0Sum) [checkI, checkIII, checkIV, checkV, checkVI]
	where
		(rH,rM,rF) = tmap transitiveClosure (r0H,r0M,r0F)
		(r0H,r0M,r0F,r0Sum) = makeGraphs ps

-- | makes graphs representing R0's for each member
-- | e.g. (h0p1 = -1 <=> h0p1 < h1p1 = 1) => h1 R h0
makeGraphs :: (Eq a, Num a) => [HChoice3] -> (BinRel, BinRel, BinRel, BinRel)
makeGraphs path = (makeR0 onlyOffdiag, makeR0 gM, makeR0 gF, makeR0 onlyDiag) 
	where
		makeR0 = fromLists . map (map (==(-1)))
		--(gH,gM,gF) = let  tupleApp (x0,x1,x2) (xs0,xs1,xs2) = (repr x0 : xs0, repr x1 : xs1, repr x2 : xs2) in foldr tupleApp ([],[],[]) path  -- OLD
		gH:gM:[gF] = map (map repr . (\(x,y,z) -> x:y:[z])) path -- NEW
		onlyDiag = let v = (M.getDiag . M.fromLists) gH in M.toLists $ M.matrix (V.length v) (V.length v) (\(i,j) -> if i == j then v V.! (i-1) else 0.0)
		onlyOffdiag = M.toLists $ M.fromLists gH - M.fromLists onlyDiag 

allElements :: BinRel -> Bool
allElements = and . V.toList . M.getMatrixAsVector

anyElement :: BinRel -> Bool
anyElement = or . V.toList . M.getMatrixAsVector

-- | Returns indices columns of the preference relation matrix which puts a restriction on the individuals
restrictionsFromSum :: BinRel -> [Int]
restrictionsFromSum r0Sum = foldr (\i a -> [i | or . V.toList $ M.getCol i r0Sum] ++ a) [] [1..M.ncols r0Sum]

-- takeCols (restrictionsFromSum r0Sum) r0Sum -- r0W ...
takeCols :: [Int] -> BinRel -> BinRel
takeCols idcs mx = M.transpose . M.fromLists . map fst $ filter (\x -> snd x `elem` idcs) $ zip (M.toLists . M.transpose $ mx) [1..]

checkI :: BinRel -> BinRel -> BinRel -> BinRel -> BinRel -> BinRel -> BinRel -> Bool
checkI r0H _ r0M _ r0F _ _ = allElements $ M.elementwise (<=) r0H (M.elementwise (||) r0M r0F)

checkIII :: BinRel -> BinRel -> BinRel -> BinRel -> BinRel -> BinRel -> BinRel -> Bool
checkIII r0H _ r0M rM r0F rF _ = consistent rM r0F && consistent rF r0M
	where
		consistent r r0 = allElements $ M.elementwise (<=) (M.elementwise (&&) r0H r) r0

-- | Logic: remove all rows for which there are no restrictions from sum (if zero rows remain - immediately conclude rational: null js)
checkIV :: BinRel -> BinRel -> BinRel -> BinRel -> BinRel -> BinRel -> BinRel -> Bool
checkIV _ _ r0M rM r0F rF r0Sum = null js || (consistent rM' r0F' && consistent rF' r0M')
	where
		consistent r r0 = not . allElements $ M.elementwise (&&) (M.elementwise (||) r0Sum' r) r0
		js = restrictionsFromSum r0Sum 
		(r0M', rM', r0F', rF', r0Sum') = (takeCols js r0M, takeCols js rM, takeCols js r0F, takeCols js rF, takeCols js r0Sum) 

-- | logic: if in a any row of the or-ed pref rels, all elements are true than this is violated
checkV :: BinRel -> BinRel -> BinRel -> BinRel -> BinRel -> BinRel -> BinRel -> Bool
checkV _ _ r0M rM r0F rF r0Sum = not . or . colAllTrue $ M.elementwise (||) (M.elementwise (||) rF rM) r0Sum
	where	
		colAllTrue = map and . M.toLists . M.transpose

checkVI :: BinRel -> BinRel -> BinRel -> BinRel -> BinRel -> BinRel -> BinRel -> Bool
checkVI r0H _ _ rM _ rF _ = allElements $ M.elementwise (<=) (M.elementwise (&&) rM rF) r0H

checkIndividualRationality :: (Eq a, Ord a, Show a, Num a) => HChoice3 -> Bool
checkIndividualRationality p = not . anyElement $ M.elementwise (&&) r (M.transpose r) -- equivalently:  not . or . V.toList . M.getDiag $ r 
	where
		r = transitiveClosure r0
		(_,_,r0,_) = makeGraphs [p,p,p] -- only the third one is considered, structure remains from choicePerPeriod and not choicePerComposition setting
		

-- | this was for the household data
checkPathIndividualRationality :: (Eq a, Ord a, Show a, Num a) => [HChoice3] -> Bool
checkPathIndividualRationality ps = consistent rM && consistent rF 
	where
		consistent r = allElements $ M.elementwise (&&) r (M.transpose r) 
		(rM, rF) = (transitiveClosure r0M, transitiveClosure r0F)
		(_,r0M,r0F,_) = makeGraphs ps
