module Continuous where

import Helpers
import Data.List (nub)
import Numeric.LinearAlgebra

data Index = Index String String 
data Value = Value Double [Double] [Double] 

instance Show Index where 
	show (Index year good) = "(" ++ year ++ ", " ++ good ++")"

instance Show Value where
	show (Value p qc qf) = show $ [p] : [map ($p) [fromIntegral . length, sum] | p <- [qc,qf]]

data PQ = PQ Index Value deriving (Show)

byYear :: [PQ] -> String -> [PQ] 
byYear pq year = filter findYear pq
	where 
		findYear (PQ (Index year' _) _) = year == year' 

sampleSize :: [PQ] -> [Int]		
sampleSize = head . map size where size (PQ _ (Value _ dc df)) = length dc : [length df] 


reshapeToMatrices :: [PQ] -> (Vector Double, Matrix Double, Matrix Double) 
reshapeToMatrices = toLA . foldr reshaper ([],[],[]) 
	where
		reshaper (PQ _ (Value p qc qf)) (ps,qcs,qfs) = (p:ps, qc:qcs, qf:qfs)
		toLA :: ([Double], [[Double]], [[Double]]) -> (Vector Double, Matrix Double, Matrix Double)
		toLA (p, qc, qf) = (fromList p, tr . fromLists $ qc, tr . fromLists $ qf) 
	
getYears :: [PQ] -> [String]
getYears = nub . map extractYear 
	where extractYear (PQ (Index year _) _) = year 

encodeBudget :: [[Double]] -> [[Double]]
encodeBudget = map (map (encodeOrd . compareFloat 1.0)) 

discretizeIndiv :: [PQ] -> String -> [[Double]]
discretizeIndiv pq year = encodeBudget $ map (toList . perBudget p qf) qs
	where 
		(p, _, qf) = (reshapeToMatrices . byYear pq) year 
		qs = map (trd3 . reshapeToMatrices . byYear pq) (getYears pq) -- ps: vs fst3

discretizeCouple :: [PQ] -> String -> [[Double]]
discretizeCouple pq year = encodeBudget $ map (toList . uncurry perBudget (pqByYear year)) qs -- replace zeros by doublesums
	where 
		 ds = doubleSums year $ filter (not . (==) year) (getYears pq)
		 doubleSums y ys = let (p, q) = (fst . pqByYear $ y, sum $ map (snd . pqByYear) ys) in q 
		 pqByYear year = let (p, qc, _) = (reshapeToMatrices . byYear pq) year in (p, qc)
		 qs = map (\y -> if y /= year then (snd3 . reshapeToMatrices . byYear pq) y else ds) (getYears pq)


perBudget :: Vector Double -> Matrix Double -> Matrix Double -> Vector Double
perBudget p q q' = totalexp
	where
		normalizedprices = cmap (1/) (q #> p) `outer` p -- vs p'
		budgetshares = normalizedprices * q' -- vs q
		totalexp = budgetshares #> fromList (replicate (size p) 1)
