-- | Check if consumption x is affordable under price regime p
checkConsumption :: [Double] -> [Double] -> Double 
checkConsumption p x = encodeOrd $ compareFloat 1.0 $ dot' p x 

-- | Constructs choice path of one individual: periods (outer), encoded polyhedra (inner)
-- | inner part refers to feasibility (checkConsumption) of choice in period t (outer) 
-- | when budget set would be s (inner)
-- | rows == cols and zero diagonal by construction
encodeConsumption :: [[Double]] -> [[Double]] -> [[Double]]
encodeConsumption ps xs = [map (uncurry checkConsumption) [(p,x) | x <- xs ] | p <- ps]

-- | Takes a vector of vectors of consumptions representing each member in the group (outer)
-- | and constructions a path of encoded group consumptions
encodeGroupConsumption :: [[Double]] -> [[[Double]]] -> [[[Double]]]
encodeGroupConsumption ps xss = transpose $ map (encodeConsumption ps) xss

-- | Returns all possible combinations of choices on in
-- | Level 0 (outermost): periods
-- | Level 1: choice set for household in respective period
-- | Level 2: particular choice of household 
-- | Level 3 (innermost): choice per individual (polyhedra)

choicesPerPeriod :: Int -> [[[IChoice]]]
choicesPerPeriod p = fmap (combinationsPeriod p) [0..(p-1)]

combinations :: Int -> [[[IChoice]]]
combinations p = sequence $ choicesPerPeriod p

combinationsPeriod :: Int -> Int -> [[IChoice]]
combinationsPeriod p t = [[IChoice h]++[IChoice m]++[IChoice f] | h<-powerset p, m <- memberchoices, f <- memberchoices]
	where
		powerset p = replicateM p [-1.0,1.0] 
		memberchoices = map (squeeze 0.0 t) (powerset (p-1))
		squeeze d t c = take t c ++ [d] ++ drop t c

choicesPerComposition :: Int -> [[[IChoice]]]
choicesPerComposition periods = [toIChoices householdchoices, toIChoices choicepaths, toIChoices choicepaths]
	where
		choices p = replicateM p [-1.0,1.0] 
		choicepaths = map (zipWith (squeeze 0.0) [0..]) $ replicateM periods (choices (periods-1))
		squeeze d t cs = take t cs ++ [d] ++ drop t cs
		householdchoices = let replace0by val = map (\x -> if x == 0 then val else x) in [zipWith replace0by ds c | ds <- choices periods, c <- choicepaths]
		toIChoices c = [[IChoice x | x <- xs] | xs <- c]
		--doublesums = map (toLists . diag . fromList) (choices periods) :: [[[Double]]]-- TODO: warning, only 3 periods supported

-- | Takes list of period choices and constructs all resulting choice combinations
-- | [[]] is unit-element of operation <*> on list applicative functor
-- | Each element is one choice path
-- | depr
combinePeriods :: [[a]] -> [[a]]
combinePeriods = foldr (\ x -> (<*>) ((:) <$> x)) [[]] 

-- | Constructs a forest from decision space and crawles the branches to check rationality
-- | once an irrational branch is found, the rationality of all child branches is set to false
-- | one-elements (start with): makeDecisionForest [] True choicespace 
makeDecisionForest :: ([HChoice3] -> Bool) -> [HChoice3] -> Bool -> [[[IChoice]]] -> Forest RChoice
makeDecisionForest _      _       _    [_] = []
makeDecisionForest checkF parents pIsR xs  = zipWith (curry makeNode) [0..] (subs xs)
	where 
		makeNode (i,x) = let c = (tuplify . head . head) x in Node (c, pIsR && isR c, i :: Int) (makeDecisionForest checkF (parents ++ [c]) (isR c) (map tail x))
		isR c = checkF (parents ++ [c])
		subs = groupBy ((==) `on` head) . sort

forestToChoicespace :: Forest RChoice -> [RChoice]
forestToChoicespace xs = join $ map (\x -> rootLabel x : forestToChoicespace (subForest x)) xs

paths :: Tree a -> [[a]] 
paths (Node label []) = [[label]]
paths (Node label xs) = map (label:) $ concatMap paths xs

classifyPaths :: Forest (HChoice3, Bool, Int) -> Paths
classifyPaths tree = map (foldr (\(l,r1,i) (ls,r2,is) -> (l:ls, r1&&r2, i:is) ) ([], True, [])) $ join (map paths tree)

asAssocMatrix :: Forest (HChoice3, Bool, Int) -> [Int] -> AssocMatrix 
asAssocMatrix forest treesplvl = join $ zipWith (curry (\(i,js) -> [((i,j), 1.0 :: Double)| j <- js])) [0..] rowIndices 
	where
		rowIndices = map (zipWith (\offset x -> offset x) (offsetters treesplvl) . trd3) rationalPaths
		rationalPaths = (filter snd3 . classifyPaths) forest

trAssocMatrix :: AssocMatrix -> AssocMatrix
trAssocMatrix = map (\((r,c),v) -> ((c,r),v))

	where

offsetters :: [Int] -> [Int -> Int]
offsetters treesplvl = fmap (+) (scanl1 (+) $ init (0:treesplvl)) --(map (ntreesplvl*) [0..])

getChoicesPerPeriod :: [Tree RChoice] -> [[RChoice]]
getChoicesPerPeriod forest = reverse $ foldr append [] [forest]
	where
		append []   current = current
		append subs current = append (subForest . head $ subs) (map rootLabel subs : current) 

lookForIndex :: (HChoice3, [RChoice]) -> Maybe Int
lookForIndex (searchterm, table) = lookup searchterm $ map (\(x,_,z) -> (x, z)) table
