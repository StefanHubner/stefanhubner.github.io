module Helpers where

-- | Basic choices
type BinRel = M.Matrix Bool -- [[Bool]]

newtype IChoice = IChoice { repr :: [Double] }
	deriving (Eq, Ord, Read, Generic)

instance NFData IChoice --where rnf x = seq x ()

type HChoice3 = (IChoice, IChoice, IChoice)
type HChoice4 = (IChoice, IChoice, IChoice, IChoice)

-- | Classified (rational/irrational) choice per period
type RChoice = (HChoice3, Bool, Int)
type Paths = [([HChoice3], Bool, [Int])]

instance Show IChoice where
	show ic = map showIndiv (repr ic)
		where
			showIndiv (-1.0) = '-'
			showIndiv  1.0 = '+'
			showIndiv  0.0 = '.'
			showIndiv  _   = '?'

fst3 :: (a,b,c) -> a
fst3 (x,_,_) = x

snd3 :: (a,b,c) -> b 
snd3 (_,y,_) = y

trd3 :: (a,b,c) -> c
trd3 (_,_,z) = z

class Tuplifyable c e where -- c = (HChoice3 a), e = [a]
	tuplify :: [e] -> c

class MapableTuple c e where
	tmap :: (e -> e) -> c -> c
	
instance MapableTuple (BinRel, BinRel, BinRel) BinRel where
	tmap f (x1,x2,x3) = (f x1, f x2, f x3)

instance MapableTuple (HChoice3, HChoice3, HChoice3) HChoice3 where
	tmap f (x1,x2,x3) = (f x1, f x2, f x3)

instance MapableTuple (BinRel, BinRel, BinRel, BinRel) BinRel where
	tmap f (x1,x2,x3,x4) = (f x1, f x2, f x3, f x4)
	
instance Tuplifyable HChoice3 IChoice where
	tuplify []                = undefined
	tuplify [_]               = undefined
	tuplify [_,_]             = undefined 
	tuplify [x,y,z]           = (x,y,z)
	tuplify (_:(_:(_:(_:_)))) = undefined 

instance Tuplifyable HChoice4 IChoice where
	tuplify []                = undefined
	tuplify [_]               = undefined
	tuplify [_,_]             = undefined 
	tuplify [_,_,_]           = undefined 
	tuplify [x1,x2,x3,x4]     = (x1,x2,x3,x4)
	tuplify (_:(_:(_:(_:(_:_))))) = undefined 

tuplify2 :: [a] -> (a,a)
tuplify2 []                = undefined
tuplify2 [_]               = undefined
tuplify2 [x,y]             = (x,y) 
tuplify2  (_:(_:(_:_)))    = undefined 

dot' :: [Double] -> [Double] -> Double	
dot' x y = sum $ zipWith (*) x y

encodeOrd :: Ordering -> Double
encodeOrd r = case r of
    LT -> 1
    GT -> -1
    EQ -> 0


boolToDouble :: Bool -> Double
boolToDouble True  = 1.0
boolToDouble False = 0.0

compareFloat :: Double -> Double -> Ordering
compareFloat f1 f2 = compare (roundTo prec f1) (roundTo prec f2)
	where
		prec = 3 
		roundTo :: Integer -> Double -> Double
		roundTo n f = fromInteger (round $ f * (10^n)) / (10.0^^n)

histogram :: (Ord a, Eq a) => [a] -> [a] -> [(a, Int)]		
histogram dat bins = let addElem m e = Data.Map.insertWith (+) e 1 m in Data.Map.toList $ Data.List.foldl addElem (Data.Map.fromList (map (flip (,) 0) bins)) dat 

safeHistogram :: (Ord a, Eq a) => [a] -> [a] -> [(a, Int)]		
safeHistogram dat bins = histogram (filter (`elem` bins) dat) bins
