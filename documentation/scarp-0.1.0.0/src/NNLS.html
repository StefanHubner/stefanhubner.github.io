module NNLS where

import Numeric.LinearAlgebra
import qualified Math.LinearAlgebra.Sparse.Matrix as S
import qualified Math.LinearAlgebra.Sparse.Vector as Sv

type StepInfoSparse = (Int, Sv.SparseVector Double) 
type StepInfo = (Int, Vector Double) 

stepLandweber :: Matrix Double -> Vector Double -> Vector Double -> StepInfo -> StepInfo
stepLandweber dmH dvdinv dvf (i,dvx) = (i-1, dvx') 
	where 
		dvmu = (dmH #> dvx) + dvf
		dvx' = cmap (max 0.0) (dvx - dvmu * dvdinv)

stepwiseOptimizer :: (StepInfo -> StepInfo) -> StepInfo -> StepInfo 
stepwiseOptimizer step (i, x) = if exitCriterion then (i', x') else stepwiseOptimizer step (i', x') 
	where
		(i', x') = step (i, x)
		exitCriterion = i' <= 0 || sqrt ((x - x') `dot` (x - x')) < 1e-6


fctJ :: Int -> Matrix Double -> Vector Double -> Vector Double -> Vector Double -> Double
fctJ n dmH dvb dvf dvNu = (fromIntegral n :: Double) * ((dvNu <# dmH) `dot` dvNu + 2 * dvf `dot` dvNu + dvb `dot` dvb)

--- | DEPREVIATED
stepLandweberSparse :: S.SparseMatrix Double -> Sv.SparseVector Double -> Sv.SparseVector Double -> StepInfoSparse -> StepInfoSparse
stepLandweberSparse smH svdinv svf (i,svx) = (i+1, svx') 
	where 
		svmu = smH `S.mulMV` svx + svf
		svx' = throwOutNegatives (length svx) $ svx - svmu*svdinv
		throwOutNegatives size = Sv.vecFromAssocListWithSize size . filter ((>0) . snd) . Sv.vecToAssocList

stepwiseOptimizerSparse :: (StepInfoSparse -> StepInfoSparse) -> StepInfoSparse -> StepInfoSparse 
stepwiseOptimizerSparse step (i, x) = if exitCriterion then (i', x') else stepwiseOptimizerSparse step (i', x') 
	where
		(i', x') = step (i, x)
		exitCriterion = i' >= 100 || sqrt ((x - x') `Sv.dot` (x - x')) < 1e-6

fctJSparse :: Int -> S.SparseMatrix Double -> Sv.SparseVector Double -> Sv.SparseVector Double -> Double
fctJSparse n smA svPi svNu = (fromIntegral n :: Double) * (smA `S.mulMV` svNu - svPi) `Sv.dot` (smA `S.mulMV` svNu - svPi)
