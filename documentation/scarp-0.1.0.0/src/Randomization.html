module Randomization where

import Control.Monad 
import Control.Monad.Primitive
import System.Random.MWC

randomChoicePath :: PrimMonad m => [a] -> Maybe [Double] -> Gen (PrimState m) -> m a
randomChoicePath pool Nothing s     = do
	index <- liftM floor (uniformR (0::Double, fromIntegral . length $ pool) s) 
	return (pool !! index)
randomChoicePath pool (Just fnu) s = do
	u <- uniformR (0,1) s
	let index = discreteQuantile (reverse $  [0.0] ++ fnu ++ [1.0]) u
	return (pool !! index) 

discreteQuantile :: [Double] -> Double -> Int 
discreteQuantile []       tau = undefined -- should never happen 
discreteQuantile (u:l:ps) tau 
	| l <= tau && tau < u = length ps
	| otherwise           = discreteQuantile (l:ps) tau 

randomChoicePath' :: PrimMonad m => [a] -> [a] -> Double -> Gen (PrimState m) -> m a
randomChoicePath' rat irr prat s = do
	u <- uniformR (0,1) s
	let rational = u < prat
	if rational then do
		index <- liftM floor (uniformR (0::Double, fromIntegral . length $ rat) s) 
		return (rat !! index)
	else do
		index <- liftM floor (uniformR (0::Double, fromIntegral . length $ irr) s)
		return (irr !! index)

bootstrapIdcs :: PrimMonad m => [Int] -> Int -> Gen (PrimState m) -> m [[[Int]]]
bootstrapIdcs ns b s = replicateM b $ mapM (\nt -> replicateM nt (liftM floor (uniformR (0::Double, fromIntegral nt :: Double) s))) ns
