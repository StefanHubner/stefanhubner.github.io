module RussianData where 

import Control.Monad
import Data.List (transpose,intersperse)

import Settings
import Continuous


priceDataRussia :: IO [((String, String), Double)]
priceDataRussia = do
	(ph:ps) <- liftM (map words . lines) . readFile $  dataPrefix ++ "/p/prices.csv"
	let (pt, p) = (map head ps, map (tail . map read) ps) :: ([String], [[Double]])
	return $ zip ((,) <$> pt <*> ph) (join p)

prepareDataset ::  [String] -> [String] -> IO [PQ]
prepareDataset goods years = do  
	let l = length goods
	let filename = joinList "_" $ joinList "-" goods : [joinList "-" years] where joinList d = filter (/=' ') .  unwords . intersperse d 
	let parseToDouble c = map (map read) c :: [[Double]]
	let readData dir = liftM (transpose . parseToDouble . map words . lines) . readFile $  dataPrefix ++ dir ++ "/" ++ filename 
	couples <- readData "q.couples"
	singles <- readData "q.singles"
	prices <- priceDataRussia
	let relevantPrices = filter ((`elem` ((,) <$> years <*> goods)) . fst) prices
	return $ zipWith3 (\((year, good), p) qcs qss -> PQ (Index year good) (Value p qcs qss)) relevantPrices couples singles
